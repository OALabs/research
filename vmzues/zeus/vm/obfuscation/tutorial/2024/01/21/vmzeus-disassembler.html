class DataSize:
    def __init__(self, value):
        self.value = value

    @classmethod
    def BYTE(cls):
        return cls(1)

    @classmethod
    def WORD(cls):
        return cls(2)

    @classmethod
    def DWORD(cls):
        return cls(4)

    def __str__(self):
        if self.value == 1:
            return 'BYTE'
        elif self.value == 2:
            return 'WORD'
        elif self.value == 4:
            return 'DWORD'
        else:
            return f'VALUE({self.value})'

class Instruction:
    def __init__(self, code):
        self.size = 0
        self.text = ""
        self.key = 0
        self.operands = []
    
    def parse(self, code):
        for operand in self.operands:
            operand.parse(code[self.size:])
            self.size += operand.op_size
            self.text += " " + operand.text
        # Assumption, the first operand is never larger than 1 byte
        if len(self.operands) == 0:
            # Key is the opcode
            self.key ^= code[0]
        elif len(self.operands) == 1:
            # Check if the operand is a real operand or a fake operand
            if self.operands[0].op_size != 0:
                # Key is the first byte in the operand
                self.key ^= code[1]
            else:
                # Key is the opcode
                self.key ^= code[0]
        elif len(self.operands) == 2:
            # If the both operands are real operands, the key is the first byte of the second operand
            if self.operands[0].op_size != 0 and self.operands[1].op_size != 0:
                self.key ^= code[1] 
            # If the first operand is real and the second is fake, the key is the first byte of the first operand
            elif self.operands[0].op_size != 0 and self.operands[1].op_size == 0:
                self.key ^= code[1]
            # If the second operand is real and the first is fake, the key is the first byte of the second operand
            elif self.operands[0].op_size == 0 and self.operands[1].op_size != 0:
                self.key ^= code[2]
            else:
                raise Exception("Invalid operands")
        elif len(self.operands) == 4:
            # There is only one case with more than 2 operands, rc4
            # key_len, data_len, key_buff[key_len], (fake)data_buff[data_len]
            # The key is the first byte in the third operand
            self.key ^= code[3]
        else:
            raise Exception("Invalid operands")

def disassemble(code):
    ptr = 0
    disassembly = []
    while ptr < len(code):
        op = code[ptr]
        if ptr > 0:
             op ^= disassembly[0].key
        disassembly.append(instructions[op](code[ptr:]))
        ptr += disassembly[-1].size
        if disassembly[-1].text == "exit":
            break
